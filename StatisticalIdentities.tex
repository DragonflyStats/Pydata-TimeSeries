\subsection{Statistical Identities}

\begin{itemize}
\item $ var[aX+b] =a^2var[X]$

\item $ var[X+Y] =var[X] +var[Y] $ provided X and Y are independent.

\item $ var[X+Y] =var[X] +var[Y] + 2cov[XY]  $

\item $ cov[XY] = E[XY] - E[X]E[Y]$

\item $ var[X] =E[X^2] -E[X]^2$
\end{itemize}
